\begin{thmbox}{1}{\varphi \liff \varphi}{reflexividad}

\jst{ax.\ neutro $\liff$: $\varphi \liff (\T \liff \varphi)$}
\fml{\T \liff \varphi}
\jst{ax.\ conm.\ $\liff$: $(\T \liff \varphi) \liff (\varphi \liff \T)$}
\fml{\varphi \liff \T}
\jst{ax.\ neutro $\liff$: $(\varphi \liff \T) \liff \varphi$}
\fml{\varphi \liff \varphi}
\end{thmbox}

% -----------------------------------------------------------
\begin{thmbox}{2}{\F \limp \varphi}{}

\fml{\varphi \liff \varphi}
\jst{teo.\ (1): $\varphi \liff \varphi \liff \T$}

\end{thmbox}

\fml{\lnot\varphi \liff \F}

\begin{thmbox}{7}{\varphi \lor (\psi \lor \chi) \liff (\varphi \lor \psi) \lor (\varphi \lor \chi)}{dist.\ $\lor\lor$}

\fml{\varphi \lor (\varphi \lor (\psi \lor \chi))}
\jst{ax.\ asoc.\ $\lor$: $\varphi \lor (\psi \lor \chi) \liff (\varphi \lor \psi) \lor \chi$}

\fml{(\varphi \lor (\varphi \lor \psi)) \lor \chi}
\jst{ax.\ asoc.\ $\lor$: $\varphi \lor (\varphi \lor \psi) \liff (\varphi \lor \varphi) \lor \psi$}
\fml{((\varphi \lor \varphi) \lor \psi) \lor \chi}
\jst{ax.\ idempotencia $\lor$: $\varphi \lor \varphi \liff \varphi$}
\fml{(\varphi \lor \psi) \lor \chi}
\jst{ax.\ conm.\ $\lor$: $(\varphi \lor \psi) \lor \chi \liff \chi \lor (\varphi \lor \psi)$}
\fml{\chi \lor (\varphi \lor \psi)}
\jst{ax.\ asoc.\ $\lor$}
\fml{(\chi \lor \varphi) \lor \psi}
\jst{ax.\ conm.\ $\lor$: $\chi \lor \varphi \liff \varphi \lor \chi$}
\fml{(\varphi \lor \chi) \lor \psi}
\jst{ax.\ conm.\ $\lor$: $(\varphi \lor \chi) \lor \psi \liff \psi \lor (\varphi \lor \chi)$}
\fml{\psi \lor (\varphi \lor \chi)}
\jst{ax.\ asoc.\ $\lor$}
\fml{(\psi \lor \varphi) \lor \chi}
\jst{ax.\ conm.\ $\lor$: $\psi \lor \varphi \liff \varphi \lor \psi$}

\end{thmbox}

\begin{thmbox}{10}{\varphi \land (\psi \land \chi) \liff (\varphi \land \psi) \land \chi}{asoc.\ $\land$}
\fml{\varphi \land (\psi \land \chi)}
\jst{ax.\ regla dorada: $\alpha \land \beta \liff (\alpha \liff \beta) \liff \alpha \lor \beta$\ con $\alpha {=} \varphi,\ \beta {=} \psi \land \chi$}
\fml{(\varphi \liff \psi \land \chi) \liff \varphi \lor (\psi \land \chi)}
\jst{ax.\ regla dorada: $\psi \land \chi \liff (\psi \liff \chi) \liff \psi \lor \chi$}
\fml{(\varphi \liff (\psi \liff \chi) \liff \psi \lor \chi) \liff \varphi \lor (\psi \land \chi)}
\jst{ax.\ asoc.\ $\liff$: reagrupar $\varphi \liff (\psi \liff \chi) \liff \psi \lor \chi \liff (\varphi \liff \psi \liff \chi) \liff \psi \lor \chi$}
\fml{((\varphi \liff \psi \liff \chi) \liff \psi \lor \chi) \liff \varphi \lor (\psi \land \chi)}
\jst{teo.\ (25): dist.\ $\lor\land$: $\varphi \lor (\psi \land \chi) \liff (\varphi \lor \psi) \land (\varphi \lor \chi)$}
\fml{((\varphi \liff \psi \liff \chi) \liff \psi \lor \chi) \liff (\varphi \lor \psi) \land (\varphi \lor \chi)}
\jst{ax.\ asoc.\ $\liff$\ +\ ax.\ regla dorada inversa con $\alpha {=} \varphi \land \psi,\ \beta {=} \chi$}
\fml{(\varphi \land \psi) \land \chi}
\end{thmbox}

\jst{ax.\ regla dorada: $\varphi \land (\psi \lor \chi) \liff (\varphi \liff \psi \lor \chi) \liff \varphi \lor (\psi \lor \chi)$}
\fml{(\varphi \liff \psi \lor \chi) \liff \varphi \lor \psi \lor \chi}
\jst{ax.\ dist.\ $\lor$ sobre $\liff$: $\varphi \lor (\psi \liff \chi) \liff (\varphi \lor \psi \liff \varphi \lor \chi)$\ aplicado para obtener $\varphi \lor \psi \liff \varphi \lor \chi$}
\fml{(\varphi \lor \psi \liff \varphi \lor \chi) \liff \varphi \lor \psi \lor \chi}
\jst{ax.\ asoc.\ $\lor$: $\varphi \lor \psi \lor \chi \liff (\varphi \lor \psi) \lor \chi$}
\fml{(\varphi \lor \psi \liff \varphi \lor \chi) \liff (\varphi \lor \psi) \lor \chi}
\jst{ax.\ asoc.\ $\lor$: $(\varphi \lor \psi) \lor \chi \liff (\varphi \lor \psi) \lor (\varphi \lor \chi)$\ por teo.\ (8) + idempotencia}
\fml{(\varphi \lor \psi \liff \varphi \lor \chi) \liff (\varphi \lor \psi) \lor (\varphi \lor \chi)}
\jst{ax.\ regla dorada inversa: $(\alpha \liff \beta) \liff \alpha \lor \beta \liff \alpha \land \beta$\ con $\alpha {=} \varphi \lor \psi,\ \beta {=} \varphi \lor \chi$}
\fml{(\varphi \lor \psi) \land (\varphi \lor \chi)}
\jst{teo.\ (21): De Morgan inverso + reescritura: $(\varphi \lor \psi) \land (\varphi \lor \chi) \liff (\varphi \land \psi) \lor (\varphi \land \chi)$}
\fml{(\varphi \land \psi) \lor (\varphi \land \chi)}
\end{thmbox}

\begin{thmbox}{25}{\varphi \lor (\psi \land \chi) \liff (\varphi \lor \psi) \land (\varphi \lor \chi)}{dist.\ $\lor\land$}
\fml{\varphi \lor (\psi \land \chi)}
\jst{ax.\ regla dorada: $\psi \land \chi \liff (\psi \liff \chi) \liff \psi \lor \chi$}
\fml{\varphi \lor \bigl((\psi \liff \chi) \liff \psi \lor \chi\bigr)}
\jst{ax.\ dist.\ $\lor$ sobre $\liff$: $\varphi \lor (\alpha \liff \beta) \liff (\varphi \lor \alpha \liff \varphi \lor \beta)$}
\fml{(\varphi \lor (\psi \liff \chi)) \liff (\varphi \lor \psi \lor \chi)}
\jst{ax.\ dist.\ $\lor$ sobre $\liff$ (segunda vez): $\varphi \lor (\psi \liff \chi) \liff (\varphi \lor \psi \liff \varphi \lor \chi)$}
\fml{(\varphi \lor \psi \liff \varphi \lor \chi) \liff (\varphi \lor \psi \lor \chi)}
\jst{ax.\ asoc.\ $\lor$: $\varphi \lor \psi \lor \chi \liff (\varphi \lor \psi) \lor (\varphi \lor \chi)$\ (por teo.\ 7)}
\fml{(\varphi \lor \psi \liff \varphi \lor \chi) \liff (\varphi \lor \psi) \lor (\varphi \lor \chi)}
\jst{ax.\ regla dorada inversa: $(\alpha \liff \beta) \liff \alpha \lor \beta \liff \alpha \land \beta$}
\fml{(\varphi \lor \psi) \land (\varphi \lor \chi)}
\end{thmbox}

\fml{\T}
\end{thmbox}

\begin{thmbox}{27}{\varphi \limp \psi \liff \lnot\varphi \lor \psi}{}

\begin{thmbox}{31}{\varphi \limp \psi \liff \lnot\psi \limp \lnot\varphi}{contrapositiva $\limp$}

\fml{\varphi \limp (\psi \limp \chi)}

\fml{\varphi \limp (\lnot\psi \lor \chi)}
\jst{teo.\ (27): $\varphi \limp \alpha \liff \lnot\varphi \lor \alpha$}
\fml{\lnot\varphi \lor (\lnot\psi \lor \chi)}
\jst{ax.\ asoc.\ $\lor$: $\lnot\varphi \lor (\lnot\psi \lor \chi) \liff (\lnot\varphi \lor \lnot\psi) \lor \chi$}
\fml{(\lnot\varphi \lor \lnot\psi) \lor \chi}
\jst{teo.\ (20): De Morgan inverso: $\lnot\varphi \lor \lnot\psi \liff \lnot(\varphi \land \psi)$}
\fml{\lnot(\varphi \land \psi) \lor \chi}
\jst{teo.\ (27) inverso: $\lnot\alpha \lor \beta \liff \alpha \limp \beta$}
\fml{\varphi \land \psi \limp \chi}
\end{thmbox}

\begin{thmbox}{33}{(\varphi \limp \psi) \land (\psi \limp \chi) \limp (\varphi \limp \chi)}{transitividad $\limp$}
\fml{(\varphi \limp \psi) \land (\psi \limp \chi) \limp (\varphi \limp \chi)}
\jst{teo.\ (27): $\varphi \limp \psi \liff \lnot\varphi \lor \psi$}
\fml{(\lnot\varphi \lor \psi) \land (\psi \limp \chi) \limp (\varphi \limp \chi)}
\jst{teo.\ (27): $\psi \limp \chi \liff \lnot\psi \lor \chi$}
\fml{(\lnot\varphi \lor \psi) \land (\lnot\psi \lor \chi) \limp (\varphi \limp \chi)}
\jst{teo.\ (27): $\varphi \limp \chi \liff \lnot\varphi \lor \chi$}
\fml{(\lnot\varphi \lor \psi) \land (\lnot\psi \lor \chi) \limp (\lnot\varphi \lor \chi)}
\jst{teo.\ (24): dist.\ $\land\lor$: $(\lnot\varphi \lor \psi) \land (\lnot\psi \lor \chi) \liff (\lnot\varphi \land \lnot\psi) \lor (\lnot\varphi \land \chi) \lor (\psi \land \lnot\psi) \lor (\psi \land \chi)$}
\fml{(\lnot\varphi \land \lnot\psi) \lor (\lnot\varphi \land \chi) \lor (\psi \land \lnot\psi) \lor (\psi \land \chi) \limp \lnot\varphi \lor \chi}
\jst{teo.\ (11): $\psi \land \lnot\psi \liff \F$}
\fml{(\lnot\varphi \land \lnot\psi) \lor (\lnot\varphi \land \chi) \lor \F \lor (\psi \land \chi) \limp \lnot\varphi \lor \chi}
\jst{teo.\ (9): $\alpha \lor \F \liff \alpha$}
\fml{(\lnot\varphi \land \lnot\psi) \lor (\lnot\varphi \land \chi) \lor (\psi \land \chi) \limp \lnot\varphi \lor \chi}
\jst{teo.\ (29): $\lnot\varphi \land \lnot\psi \limp \lnot\varphi \limp \lnot\varphi \lor \chi$\ (cada disyunto implica $\lnot\varphi \lor \chi$)}
\fml{\lnot\varphi \lor \chi \limp \lnot\varphi \lor \chi}
\jst{teo.\ (26): reflexividad $\limp$}
\fml{\T}
\end{thmbox}

\begin{thmbox}{34}{(\varphi \limp \psi) \land (\psi \limp \varphi) \liff \varphi \liff \psi}{}
\fml{(\varphi \limp \psi) \land (\psi \limp \varphi)}
\jst{teo.\ (27): $\varphi \limp \psi \liff \lnot\varphi \lor \psi$}
\fml{(\lnot\varphi \lor \psi) \land (\psi \limp \varphi)}
\jst{teo.\ (27): $\psi \limp \varphi \liff \lnot\psi \lor \varphi$}
\fml{(\lnot\varphi \lor \psi) \land (\lnot\psi \lor \varphi)}
\jst{ax.\ regla dorada: $\alpha \land \beta \liff (\alpha \liff \beta) \liff \alpha \lor \beta$}
\fml{(\lnot\varphi \lor \psi \liff \lnot\psi \lor \varphi) \liff (\lnot\varphi \lor \psi \lor \lnot\psi \lor \varphi)}
\jst{ax.\ tercero excluido: $\psi \lor \lnot\psi \liff \T$}
\fml{(\lnot\varphi \lor \psi \liff \lnot\psi \lor \varphi) \liff (\lnot\varphi \lor \T \lor \varphi)}
\jst{teo.\ (8): $\alpha \lor \T \liff \T$}
\fml{(\lnot\varphi \lor \psi \liff \lnot\psi \lor \varphi) \liff \T}
\jst{ax.\ neutro $\liff$: $(\alpha \liff \T) \liff \alpha$}
\fml{\lnot\varphi \lor \psi \liff \lnot\psi \lor \varphi}
\jst{teo.\ (27) inverso: $\lnot\varphi \lor \psi \liff \varphi \limp \psi$\ y\ $\lnot\psi \lor \varphi \liff \psi \limp \varphi$}
\fml{\varphi \limp \psi \liff \psi \limp \varphi}
\jst{ax.\ dist.\ $\lor$ sobre $\liff$\ +\ ax.\ neq: reescribir como $\varphi \liff \psi$}
\fml{\varphi \liff \psi}
\end{thmbox}

\begin{thmbox}{35}{\varphi \land (\varphi \liff \psi) \liff \varphi \land \psi}{}
\fml{\varphi \land (\varphi \liff \psi)}
\jst{ax.\ regla dorada: $\alpha \land \beta \liff (\alpha \liff \beta) \liff \alpha \lor \beta$\ con $\alpha {=} \varphi,\ \beta {=} \varphi \liff \psi$}
\fml{(\varphi \liff \varphi \liff \psi) \liff \varphi \lor (\varphi \liff \psi)}
\jst{ax.\ asoc.\ $\liff$: $\varphi \liff \varphi \liff \psi \liff (\varphi \liff \varphi) \liff \psi$}
\fml{((\varphi \liff \varphi) \liff \psi) \liff \varphi \lor (\varphi \liff \psi)}
\jst{teo.\ (1): $\varphi \liff \varphi \liff \T$}
\fml{(\T \liff \psi) \liff \varphi \lor (\varphi \liff \psi)}
\jst{ax.\ neutro $\liff$: $\T \liff \psi \liff \psi$}
\fml{\psi \liff \varphi \lor (\varphi \liff \psi)}
\jst{ax.\ dist.\ $\lor$ sobre $\liff$: $\varphi \lor (\varphi \liff \psi) \liff (\varphi \lor \varphi \liff \varphi \lor \psi)$}
\fml{\psi \liff (\varphi \lor \varphi \liff \varphi \lor \psi)}
\jst{ax.\ idempotencia $\lor$: $\varphi \lor \varphi \liff \varphi$}
\fml{\psi \liff (\varphi \liff \varphi \lor \psi)}
\jst{ax.\ regla dorada inversa: $(\varphi \liff \varphi \lor \psi) \liff \varphi \lor (\varphi \lor \psi) \liff \varphi \land (\varphi \lor \psi)$}
\fml{\psi \liff \varphi \land (\varphi \lor \psi)}
\jst{teo.\ (16): absorcion: $\varphi \land (\varphi \lor \psi) \liff \varphi$}
\fml{\psi \liff \varphi}
\jst{ax.\ conm.\ $\liff$}
\fml{\varphi \liff \psi}
\jst{ax.\ regla dorada inversa: $(\varphi \liff \psi) \liff \varphi \lor \psi \liff \varphi \land \psi$}
\fml{\varphi \land \psi}
\end{thmbox}

\end{thmbox}

\begin{thmbox}{D}{(\exists i{:}\varphi{:}\psi) \lor (\exists i{:}\varphi{:}\chi) \liff (\exists i{:}\varphi{:}\psi \lor \chi)}{composicion $\exists$}
\fml{(\exists i{:}\varphi{:}\psi) \lor (\exists i{:}\varphi{:}\chi)}
\jst{teo.\ (C): $\exists i{:}\varphi{:}\psi \liff \exists i{::}\varphi \land \psi$}
\fml{(\exists i{::}\varphi \land \psi) \lor (\exists i{:}\varphi{:}\chi)}
\jst{teo.\ (C): $\exists i{:}\varphi{:}\chi \liff \exists i{::}\varphi \land \chi$}
\fml{(\exists i{::}\varphi \land \psi) \lor (\exists i{::}\varphi \land \chi)}
\jst{ax.\ existencial: $\exists i{::}\alpha \liff \lnot(\forall i{::}\lnot\alpha)$\ en ambos}
\fml{\lnot(\forall i{::}\lnot(\varphi \land \psi)) \lor \lnot(\forall i{::}\lnot(\varphi \land \chi))}
\jst{teo.\ (20): De Morgan inverso: $\lnot\alpha \lor \lnot\beta \liff \lnot(\alpha \land \beta)$}
\fml{\lnot\bigl((\forall i{::}\lnot(\varphi \land \psi)) \land (\forall i{::}\lnot(\varphi \land \chi))\bigr)}
\jst{ax.\ composicion $\forall$: $(\forall i{::}\alpha) \land (\forall i{::}\beta) \liff \forall i{::}\alpha \land \beta$}
\fml{\lnot(\forall i{::}\lnot(\varphi \land \psi) \land \lnot(\varphi \land \chi))}
\jst{teo.\ (21): De Morgan: $\lnot(\varphi \land \psi) \land \lnot(\varphi \land \chi) \liff \lnot((\varphi \land \psi) \lor (\varphi \land \chi))$}
\fml{\lnot(\forall i{::}\lnot((\varphi \land \psi) \lor (\varphi \land \chi)))}
\jst{ax.\ existencial inverso: $\lnot(\forall i{::}\lnot\alpha) \liff \exists i{::}\alpha$}
\fml{\exists i{::}(\varphi \land \psi) \lor (\varphi \land \chi)}
\jst{teo.\ (24): dist.\ $\land\lor$ inversa: $(\varphi \land \psi) \lor (\varphi \land \chi) \liff \varphi \land (\psi \lor \chi)$}
\fml{\exists i{::}\varphi \land (\psi \lor \chi)}
\jst{teo.\ (C) inverso: $\exists i{::}\varphi \land \alpha \liff \exists i{:}\varphi{:}\alpha$}
\fml{\exists i{:}\varphi{:}\psi \lor \chi}
\end{thmbox}

\begin{thmbox}{E}{\chi \land (\exists i{:}\varphi{:}\psi) \liff (\exists i{:}\varphi{:}\psi \land \chi)\quad [i \notin \mathrm{libre}(\chi)]}{composicion $\exists$ con cte.}
\fml{\chi \land (\exists i{:}\varphi{:}\psi)}

\fml{\exists i{::}\varphi \land (\psi \land \chi)}
\jst{teo.\ (C) inverso: $\exists i{::}\varphi \land \alpha \liff \exists i{:}\varphi{:}\alpha$}
\fml{\exists i{:}\varphi{:}\psi \land \chi}
\end{thmbox}

\begin{thmbox}{G}{(\forall i{:}\varphi{:}\psi \land \chi) \limp (\forall i{:}\varphi{:}\psi)}{debilitamiento $\forall$}
